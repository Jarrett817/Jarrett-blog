TS 的类型 #

let a: any = 'hi';
// a.name

// unkown需要用断言
// let b:unknown=JSON.parse('{name:'jarrett'}')
(b as { name: string }).name;

const print: () => void = function () {};

// 空集，什么都没有
const c: never = 12;

// 枚举，一般不用，前端用 type=东|南|西|北 更香

  南,
  西,
  北,
}

let d: Dir = Dir.东; //即Dir.东是0

  if (typeof a === 'number' && typeof b === 'number') {
    return a + b;
  } else {
    return a + '' + b;
